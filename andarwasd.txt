#!/usr/bin/env python3
import rclpy
from rclpy.node import Node
from geometry_msgs.msg import Twist
import sys, select, termios, tty
import time  # Importado no topo para ser global

msg_instrucoes = """
---------------------------
Controle Manual do JetAuto
---------------------------
W : Frente
S : Trás
A : Girar Esquerda
D : Girar Direita
Espaço : PARAR

CTRL+C para sair
---------------------------
"""

class TeleopJetAuto(Node):
    def __init__(self):
        super().__init__('teleop_manual_node')
        self.pub = self.create_publisher(Twist, '/cmd_vel', 10)
        self.settings = termios.tcgetattr(sys.stdin)

    def get_key(self):
        tty.setraw(sys.stdin.fileno())
        select.select([sys.stdin], [], [], 0.1)
        key = sys.stdin.read(1)
        termios.tcsetattr(sys.stdin, termios.TCSADRAIN, self.settings)
        return key

    def enviar_comando(self, vx, az):
        msg = Twist()
        msg.linear.x = float(vx)
        msg.angular.z = float(az)
        self.pub.publish(msg)

    def parar_robusto(self):
        msg = Twist()
        msg.linear.x = 0.0
        msg.angular.z = 0.0
        # Agora o 'time' será reconhecido aqui
        for _ in range(5):
            self.pub.publish(msg)
            time.sleep(0.05)

    def executar(self):
        vel_linear = 0.2
        vel_angular = 1.0
        
        print(msg_instrucoes)

        try:
            while rclpy.ok():
                key = self.get_key()
                if key == 'w':
                    self.enviar_comando(vel_linear, 0.0)
                elif key == 's':
                    self.enviar_comando(-vel_linear, 0.0)
                elif key == 'a':
                    self.enviar_comando(0.0, vel_angular)
                elif key == 'd':
                    self.enviar_comando(0.0, -vel_angular)
                elif key == ' ':
                    self.parar_robusto()
                elif key == '\x03': # CTRL+C
                    break
                
        except Exception as e:
            print(f"\nErro detectado: {e}")
        finally:
            self.parar_robusto()

def main(args=None):
    rclpy.init(args=args)
    node = TeleopJetAuto()
    try:
        node.executar()
    finally:
        node.destroy_node()
        rclpy.shutdown()

if __name__ == '__main__':
    main()
