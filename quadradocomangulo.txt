#!/usr/bin/env python3
import rclpy
from rclpy.node import Node
from geometry_msgs.msg import Twist
import time

class JetAutoSquare(Node):

    def __init__(self):
        super().__init__('jetauto_square')
        self.pub = self.create_publisher(Twist, '/cmd_vel', 10)

    def andar(self, velocidade, tempo):
        msg = Twist()
        inicio = time.time()

        while time.time() - inicio < tempo:
            msg.linear.x = velocidade
            msg.angular.z = 0.0
            self.pub.publish(msg)
            time.sleep(0.05)

    def girar(self, velocidade_angular, tempo):
        msg = Twist()
        inicio = time.time()

        while time.time() - inicio < tempo:
            msg.linear.x = 0.0
            msg.angular.z = velocidade_angular
            self.pub.publish(msg)
            time.sleep(0.05)

    def parar(self):
        msg = Twist()
        for _ in range(5):
            msg.linear.x = 0.0
            msg.angular.z = 0.0
            self.pub.publish(msg)
            time.sleep(0.05)

    def executar(self):
        # parâmetros
        velocidade = 0.1
        lado = 0.2  # 20 cm
        tempo_reto = lado / velocidade

        vel_angular = 1.0
        tempo_giro = 1.57 / vel_angular  # ~90 graus

        time.sleep(1)

        for _ in range(2):  # 2 quadrados
            for _ in range(4):  # 4 lados
                self.andar(velocidade, tempo_reto)
                self.parar()

                self.girar(vel_angular, tempo_giro)
                self.parar()

        self.parar()


def main():
    rclpy.init()
    node = JetAutoSquare()

    node.executar()

    node.destroy_node()
    rclpy.shutdown()


if __name__ == '__main__':
    main()
