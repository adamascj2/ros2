#!/usr/bin/env python3
import rclpy
from rclpy.node import Node
from sensor_msgs.msg import LaserScan
from geometry_msgs.msg import Twist
from rclpy.qos import QoSProfile, ReliabilityPolicy, HistoryPolicy
import time

class JetAutoFinal(Node):
    def __init__(self):
        super().__init__('jetauto_final')
        
        # Configuração QoS validada para o Jeti-Auto
        qos = QoSProfile(
            reliability=ReliabilityPolicy.BEST_EFFORT,
            history=HistoryPolicy.KEEP_LAST,
            depth=10
        )

        self.publisher_ = self.create_publisher(Twist, '/cmd_vel', 10)
        self.subscription = self.create_subscription(LaserScan, '/scan', self.scan_callback, qos)
        
        self.distancia_frontal = 99.0
        self.running = True
        self.tempo_limite = 15.0
        self.distancia_stop = 0.30  # Segurança: 30 centímetros
        
        print(f"{'TEMPO':<10} | {'LIDAR (FRENTE)':<15} | {'MOTORES'}")
        print("-" * 45)
        
        self.start_time = time.time()
        # Loop de controle a 10Hz (sincronizado com o Lidar)
        self.timer = self.create_timer(0.1, self.control_loop)

    def scan_callback(self, msg):
        """ Analisa a 'fatia' frontal do LIDAR (Índices iniciais e finais) """
        # O 'zero' do Lidar A1 costuma ficar dividido entre o início e fim do array
        puntos = [d for d in msg.ranges[:15] if d > 0.05] + \
                [d for d in msg.ranges[-15:] if d > 0.05]
        
        if puntos:
            # Pegamos o valor mínimo da fatia frontal para segurança máxima
            self.distancia_frontal = min(puntos)

    def control_loop(self):
        if not self.running:
            return

        decorrido = time.time() - self.start_time
        
        # Log para a gravação do vídeo
        status = "ATIVOS"
        if self.distancia_frontal < self.distancia_stop:
            status = "!!! STOP !!!"
        
        print(f"{decorrido:>8.1f}s | {self.distancia_frontal:>11.3f} m | {status}")

        # Lógica de Decisão (Navegação Reativa)
        if self.distancia_frontal < self.distancia_stop:
            self.desligar_robo("OBSTACULO DETECTADO")
        elif decorrido >= self.tempo_limite:
            self.desligar_robo("TEMPO ESGOTADO")
        else:
            # Comando de movimento linear constante (10cm/s)
            cmd = Twist()
            cmd.linear.x = 0.1 
            self.publisher_.publish(cmd)

    def desligar_robo(self, razao):
        self.running = False
        cmd = Twist()
        cmd.linear.x = 0.0
        # Envia múltiplos pacotes de 'zero' para garantir a parada física
        for _ in range(10):
            self.publisher_.publish(cmd)
        
        print("-" * 45)
        print(f"SISTEMA ENCERRADO: {razao}")
        self.timer.cancel()
        raise SystemExit

def main():
    rclpy.init()
    node = JetAutoFinal()
    try:
        rclpy.spin(node)
    except (KeyboardInterrupt, SystemExit):
        pass
    finally:
        node.destroy_node()
        rclpy.shutdown()

if __name__ == '__main__':
    main()
