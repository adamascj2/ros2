#!/usr/bin/env python3
import os
import time
import speech_recognition as sr

# Seu microfone USB
MIC_INDEX = 26

def falar(texto):
    print(f"[ROBOT] {texto}")
    comando = f'espeak -v pt-br "{texto}" --stdout | aplay -D plughw:3,0'
    os.system(comando)

def ouvir():
    rec = sr.Recognizer()
    rec.energy_threshold = 300

    try:
        with sr.Microphone(device_index=MIC_INDEX) as source:
            print("\n🎤 Fale algo...")

            rec.adjust_for_ambient_noise(source, duration=0.5)
            audio = rec.listen(source, timeout=5, phrase_time_limit=4)

            print("Processando...")
            texto = rec.recognize_google(audio, language="pt-BR")
            texto = texto.strip()

            print(f"[VOCÊ] {texto}")
            return texto

    except sr.WaitTimeoutError:
        print("[AVISO] Ninguém falou")
    except sr.UnknownValueError:
        print("[AVISO] Não entendi")
    except Exception as e:
        print(f"[ERRO] {e}")

    return None

def main():
    print("=== ROBÔ ECO DE VOZ ===")
    falar("Pronto para repetir o que você disser")

    while True:
        texto = ouvir()

        if texto:
            falar(texto)

        time.sleep(1)

if __name__ == "__main__":
    # garante volume
    os.system("amixer sset 'Capture' 100% > /dev/null 2>&1")

    main()
